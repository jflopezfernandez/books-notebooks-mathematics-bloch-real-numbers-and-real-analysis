\chapter{Functions}
\begin{definition}\cite{bloch_proofs_2011}
	\label{definition-function}
	Let $A$ and $B$ be sets. A \textbf{function} (also called a \textbf{map}) $f$ from $A$ to $B$, denoted $f : A \to B$, is a subset $F \subseteq A \times B$ such that for each $a \in A$, there is one and only one pair in $F$ of the form $\left( a, b \right)$. The set $A$ is called the \textbf{domain}\index{domain} of $f$ and the set $B$ is called the \textbf{codomain}\index{codomain} of $f$.
\end{definition}
